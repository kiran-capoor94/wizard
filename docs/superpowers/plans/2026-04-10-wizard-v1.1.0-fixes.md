Remove `KrispClient` entirely. Rewrite `NotionClient` with notion-sdk-py. Add httpx timeout to JiraClient:

```python
import logging
import re
import httpx
from notion_client import Client as NotionSdkClient

logger = logging.getLogger(__name__)

_HTTPX_TIMEOUT = httpx.Timeout(10.0)


class ConfigurationError(Exception):
    pass


class JiraClient:
    def __init__(self, base_url: str, token: str, project_key: str):
        self._base_url = base_url.rstrip("/")
        self._token = token
        self._project_key = project_key

    def _headers(self) -> dict:
        return {
            "Authorization": f"Bearer {self._token}",
            "Content-Type": "application/json",
        }

    def fetch_open_tasks(self) -> list[dict]:
        if not self._token:
            raise ConfigurationError("Jira token not configured")
        jql = f"project={self._project_key} AND statusCategory != Done ORDER BY priority DESC"
        url = f"{self._base_url}/rest/api/2/search"
        response = httpx.get(
            url, params={"jql": jql, "maxResults": 50},
            headers=self._headers(), timeout=_HTTPX_TIMEOUT,
        )
        response.raise_for_status()
        issues = response.json().get("issues", [])
        return [
            {
                "key": issue["key"],
                "summary": issue["fields"]["summary"],
                "status": issue["fields"]["status"]["name"],
                "priority": issue["fields"]["priority"]["name"],
                "issue_type": issue["fields"]["issuetype"]["name"],
                "url": issue["fields"].get("self", ""),
            }
            for issue in issues
        ]

    def update_task_status(self, source_id: str, status: str) -> bool:
        if not self._token:
            return False
        url = f"{self._base_url}/rest/api/2/issue/{source_id}/transitions"
        try:
            response = httpx.post(
                url, json={"transition": {"name": status}},
                headers=self._headers(), timeout=_HTTPX_TIMEOUT,
            )
            response.raise_for_status()
            return True
        except httpx.HTTPError:
            return False


def _extract_jira_key(url: str) -> str | None:
    """Extract Jira issue key from a URL like https://org.atlassian.net/browse/ENG-123."""
    if not url:
        return None
    match = re.search(r"/browse/([A-Z]+-\d+)", url)
    return match.group(1) if match else None


def _get_title(properties: dict, key: str) -> str:
    """Extract plain text from a Notion title property."""
    title_list = properties.get(key, {}).get("title", [])
    return title_list[0]["plain_text"] if title_list else ""


def _get_rich_text(properties: dict, key: str) -> str:
    """Extract plain text from a Notion rich_text property."""
    rt_list = properties.get(key, {}).get("rich_text", [])
    return rt_list[0]["plain_text"] if rt_list else ""


def _get_select(properties: dict, key: str) -> str | None:
    """Extract name from a Notion select property."""
    sel = properties.get(key, {}).get("select")
    return sel["name"] if sel else None


def _get_multi_select(properties: dict, key: str) -> list[str]:
    """Extract names from a Notion multi_select property."""
    return [item["name"] for item in properties.get(key, {}).get("multi_select", [])]


def _get_url(properties: dict, key: str) -> str | None:
    """Extract URL from a Notion url property."""
    return properties.get(key, {}).get("url")


def _get_date_start(properties: dict, key: str) -> str | None:
    """Extract start date from a Notion date property."""
    date_obj = properties.get(key, {}).get("date")
    return date_obj["start"] if date_obj else None


def _get_status(properties: dict, key: str) -> str | None:
    """Extract name from a Notion status property."""
    status = properties.get(key, {}).get("status")
    return status["name"] if status else None


class NotionClient:
    def __init__(self, token: str, daily_page_id: str, tasks_db_id: str, meetings_db_id: str):
        self._token = token
        self._daily_page_id = daily_page_id
        self._tasks_db_id = tasks_db_id
        self._meetings_db_id = meetings_db_id
        self._client = NotionSdkClient(auth=token) if token else None

    def _ensure_client(self):
        if not self._client:
            raise ConfigurationError("Notion token not configured")

    # --- Read methods ---

    def fetch_tasks(self) -> list[dict]:
        self._ensure_client()
        if not self._tasks_db_id:
            return []
        results = []
        cursor = None
        while True:
            kwargs = {"database_id": self._tasks_db_id, "page_size": 100}
            if cursor:
                kwargs["start_cursor"] = cursor
            response = self._client.databases.query(**kwargs)
            for page in response["results"]:
                props = page["properties"]
                jira_url = _get_url(props, "Jira")
                results.append({
                    "notion_id": page["id"],
                    "name": _get_title(props, "Task"),
                    "status": _get_status(props, "Status"),
                    "priority": _get_select(props, "Priority"),
                    "due_date": _get_date_start(props, "Due date"),
                    "jira_url": jira_url,
                    "jira_key": _extract_jira_key(jira_url) if jira_url else None,
                })
            if not response.get("has_more"):
                break
            cursor = response.get("next_cursor")
        return results

    def fetch_meetings(self) -> list[dict]:
        self._ensure_client()
        if not self._meetings_db_id:
            return []
        results = []
        cursor = None
        while True:
            kwargs = {"database_id": self._meetings_db_id, "page_size": 100}
            if cursor:
                kwargs["start_cursor"] = cursor
            response = self._client.databases.query(**kwargs)
            for page in response["results"]:
                props = page["properties"]
                results.append({
                    "notion_id": page["id"],
                    "title": _get_title(props, "Meeting name"),
                    "categories": _get_multi_select(props, "Category"),
                    "summary": _get_rich_text(props, "Summary"),
                    "krisp_url": _get_url(props, "Krisp URL"),
                    "date": _get_date_start(props, "Date"),
                })
            if not response.get("has_more"):
                break
            cursor = response.get("next_cursor")
        return results

    # --- Write methods ---

    def create_task_page(self, name: str, status: str, priority: str,
                         jira_url: str | None = None, due_date: str | None = None) -> str | None:
        self._ensure_client()
        if not self._tasks_db_id:
            return None
        properties = {
            "Task": {"title": [{"text": {"content": name}}]},
            "Status": {"status": {"name": status}},
            "Priority": {"select": {"name": priority}},
        }
        if jira_url:
            properties["Jira"] = {"url": jira_url}
        if due_date:
            properties["Due date"] = {"date": {"start": due_date}}
        try:
            result = self._client.pages.create(
                parent={"database_id": self._tasks_db_id},
                properties=properties,
            )
            return result["id"]
        except Exception as e:
            logger.warning("Notion create_task_page failed: %s", e)
            return None

    def create_meeting_page(self, title: str, category: str,
                            krisp_url: str | None = None, summary: str | None = None) -> str | None:
        self._ensure_client()
        if not self._meetings_db_id:
            return None
        properties = {
            "Meeting name": {"title": [{"text": {"content": title}}]},
            "Category": {"multi_select": [{"name": category}]},
        }
        if krisp_url:
            properties["Krisp URL"] = {"url": krisp_url}
        if summary:
            properties["Summary"] = {"rich_text": [{"text": {"content": summary}}]}
        try:
            result = self._client.pages.create(
                parent={"database_id": self._meetings_db_id},
                properties=properties,
            )
            return result["id"]
        except Exception as e:
            logger.warning("Notion create_meeting_page failed: %s", e)
            return None

    def update_task_status(self, page_id: str, status: str) -> bool:
        self._ensure_client()
        try:
            self._client.pages.update(
                page_id=page_id,
                properties={"Status": {"status": {"name": status}}},
            )
            return True
        except Exception as e:
            logger.warning("Notion update_task_status failed: %s", e)
            return False

    def update_meeting_summary(self, page_id: str, summary: str) -> bool:
        self._ensure_client()
        try:
            self._client.pages.update(
                page_id=page_id,
                properties={"Summary": {"rich_text": [{"text": {"content": summary}}]}},
            )
            return True
        except Exception as e:
            logger.warning("Notion update_meeting_summary failed: %s", e)
            return False

    def update_daily_page(self, summary: str) -> bool:
        if not self._client or not self._daily_page_id:
            return False
        try:
            self._client.pages.update(
                page_id=self._daily_page_id,
                properties={"Session Summary": {"rich_text": [{"text": {"content": summary}}]}},
            )
            return True
        except Exception as e:
            logger.warning("Notion update_daily_page failed: %s", e)
            return False
```

class WriteBackService:
    def __init__(self, jira: JiraClient, notion: NotionClient):
        self._jira = jira
        self._notion = notion

    def push_task_status(self, task) -> bool:
        if not task.source_id:
            return False
        jira_status = _LOCAL_TO_JIRA_STATUS.get(task.status)
        if not jira_status:
            return False
        try:
            return self._jira.update_task_status(task.source_id, jira_status)
        except Exception as e:
            logger.warning("WriteBack push_task_status (Jira) failed: %s", e)
            return False

    def push_task_status_to_notion(self, task) -> bool:
        if not task.notion_id:
            return False
        notion_status = _LOCAL_TO_NOTION_STATUS.get(task.status)
        if not notion_status:
            return False
        try:
            return self._notion.update_task_status(task.notion_id, notion_status)
        except Exception as e:
            logger.warning("WriteBack push_task_status (Notion) failed: %s", e)
            return False

    def push_task_to_notion(self, task) -> bool:
        if task.notion_id:
            return self.push_task_status_to_notion(task)
        notion_status = _LOCAL_TO_NOTION_STATUS.get(task.status, "Not started")
        priority_map = {TaskPriority.HIGH: "High", TaskPriority.MEDIUM: "Medium", TaskPriority.LOW: "Low"}
        priority = priority_map.get(task.priority, "Medium")
        try:
            page_id = self._notion.create_task_page(
                name=task.name, status=notion_status, priority=priority,
                jira_url=task.source_url,
                due_date=task.due_date.isoformat() if task.due_date else None,
            )
            if page_id:
                task.notion_id = page_id
                return True
            return False
        except Exception as e:
            logger.warning("WriteBack push_task_to_notion failed: %s", e)
            return False

    def push_meeting_to_notion(self, meeting) -> bool:
        if meeting.notion_id:
            return self.push_meeting_summary(meeting) if meeting.summary else True
        category_map = {
            MeetingCategory.STANDUP: "Standup", MeetingCategory.PLANNING: "Planning",
            MeetingCategory.RETRO: "Retro", MeetingCategory.GENERAL: "Planning",
        }
        category = category_map.get(meeting.category, "Planning")
        try:
            page_id = self._notion.create_meeting_page(
                title=meeting.title, category=category,
                krisp_url=meeting.source_url, summary=meeting.summary,
            )
            if page_id:
                meeting.notion_id = page_id
                return True
            return False
        except Exception as e:
            logger.warning("WriteBack push_meeting_to_notion failed: %s", e)
            return False

    def push_meeting_summary(self, meeting) -> bool:
        if not meeting.notion_id or not meeting.summary:
            return False
        try:
            return self._notion.update_meeting_summary(meeting.notion_id, meeting.summary)
        except Exception as e:
            logger.warning("WriteBack push_meeting_summary failed: %s", e)
            return False

    def push_session_summary(self, session) -> bool:
        if not session.summary:
            return False
        try:
            return self._notion.update_daily_page(session.summary)
        except Exception as e:
            logger.warning("WriteBack push_session_summary failed: %s", e)
            return False

def ingest_meeting(
    title: str,
    content: str,
    source_id: str | None = None,
    source_url: str | None = None,
    category: str = "general",
) -> dict:
    """Accepts meeting data (e.g. from Krisp MCP), scrubs, stores, writes to Notion."""
    from .models import Meeting, MeetingCategory
    from .schemas import IngestMeetingResponse

    _sync, wb, _repo, security = _get_deps()
    db = _get_db()
    try:
        clean_title = security.scrub(title).clean
        clean_content = security.scrub(content).clean

        already_existed = False
        meeting = None
        if source_id:
            meeting = db.exec(select(Meeting).where(Meeting.source_id == source_id)).first()

        if meeting:
            already_existed = True
            meeting.title = clean_title
            meeting.content = clean_content
            db.add(meeting)
        else:
            meeting = Meeting(
                title=clean_title,
                content=clean_content,
                source_id=source_id,
                source_type="KRISP" if source_id else None,
                source_url=source_url,
                category=MeetingCategory(category),
            )
            db.add(meeting)

        db.commit()
        db.refresh(meeting)

        wb.push_meeting_to_notion(meeting)
        db.commit()

        return IngestMeetingResponse(
            meeting_id=meeting.id,
            already_existed=already_existed,
        ).model_dump(mode="json")
    finally:
        db.close()
